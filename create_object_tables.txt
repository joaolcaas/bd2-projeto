create or replace type dim_professor as object(
    id_professor INTEGER,
    nome varchar2(15)
);
/
create or replace type dim_disciplina as object(
    id_disciplina INTEGER,
    nome varchar2(60),
    sigla varchar2(6),
    area varchar2(6),
    carga INTEGER,
    creditos INTEGER
);
/

create or replace type dim_curso as object(
    id_curso INTEGER,
    nome varchar(45)
);
/

create or replace type dim_sala as object(
    id_sala INTEGER,
    codigo varchar2(8),
    capacidade INTEGER
);
/

create or replace type fat_alocacao as object(
    id_disciplina ref dim_disciplina,
    id_professor ref dim_professor,
    id_curso ref dim_curso,
    cod_turma INTEGER,
    id_sala ref dim_sala,
    periodo varchar2(6),
    qtde_alunos INTEGER,
    media_turma NUMBER(4,2)
);
